 233         /**
 234          * Stores profiling information about query execution time
 235          * \param $time_started is microtime from when the execution of this query begun
 236          * \param $q is the query being profiled
 237          * \return nothing
 238          */
 239         function profileQuery($time_started, $q)
 240         {
 241                 $this->time_spent[ $this->queries_cnt ] = microtime(true) - $time_started;
 242                 $this->queries[ $this->queries_cnt ] = $q;
 243                 $this->queries_cnt++;
 244         }

 247          * Stores profiling information about a failed query execution

 251          * \return nothing

 264         function showProfile($pageload_start = 0)

 337         {
 338                 global $session, $config;

 340                 if (!$session->isAdmin) return false;

 356                 }
 357 
 358                 $list = $this->getArray($q);
 359                 echo count($list).' entries in event log.<br/><br/>';
 360 
 361                 foreach ($list as $row)

 418         /**
 419          * Compares two MySQL datetime timestamps
 420          * \param $d1 is a MySQL datetime
 421          * \param $d2 is a MySQL datetime
 422          * \return true if $d1 is older date than $d2
