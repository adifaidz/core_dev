  20         function addComment($_type, $ownerId, $commentText, $privateComment = false)
  21         {
  22                 global $db, $session;
  23                 if (!is_numeric($_type) || !is_numeric($ownerId) || !is_bool($privateComment)) return false;
  24 
  25                 if ($_type != COMMENT_IMAGE && $_type != COMMENT_PASTEBIN && !$session->id) return false;
  26 
  27                 $commentText = $db->escape(htmlspecialchars($commentText));
  28 
  29                 if ($privateComment) $private = 1;
  30                 else $private = 0;
  31 
  32                 $q = 'INSERT INTO tblComments SET ownerId='.$ownerId.', userId='.$session->id.', userIP='.IPv4_to_GeoIP($_SERVER['REMOTE_ADDR']).', commentType='.$_type.', commentText="'.$commentText.'", commentPrivate='.$private.', timeCreated=NOW()';
  33                 return $db->insert($q);

  70 
  71                 $q  = 'SELECT t1.*,t2.userName FROM tblComments AS t1 ';
  72                 $q .= 'LEFT JOIN tblUsers AS t2 ON (t1.userId=t2.userId) ';

  81         /* Returns all comments to all objects owned by $ownerId, newest first */
  82         function getCommentsByOwner($_type, $ownerId)
  83         {
  84                 global $db, $files;

  86 
  87                 $q  = 'SELECT t1.*,t2.userName FROM tblComments AS t1 ';
  88                 $q .= 'LEFT JOIN tblUsers AS t2 ON (t1.userId=t2.userId) ';
  89                 $q .= 'WHERE t1.commentType='.$_type.' AND t1.deletedBy=0 ORDER BY t1.timeCreated DESC';
  90                 $list = $db->getArray($q);
  91 
  92                 $result = array();
  93                 foreach ($list as $row) {
  94                         if ($_type == COMMENT_IMAGE && $files->getUploader($row['ownerId']) == $ownerId) {

 128         /* Helper function, standard "show comments" to be used by other modules */
 129         //col_w sets the column width of the textarea
 130         function showComments($_type, $ownerId, $col_w = 30, $col_h = 6)
 131         {
 132                 global $session, $config;
 133                 if (!is_numeric($_type) || !is_numeric($ownerId) || !is_numeric($col_w) || !is_numeric($col_h)) return false;
 134 
 135                 if (!empty($_POST['cmt'])) {
 136                         addComment($_type, $ownerId, $_POST['cmt']);
 137                 }
 138 
 139                 if (!empty($_GET['delete']) && is_numeric($_GET['delete'])) {
 140                         deleteComment($_GET['delete']);
 141                 }
 142 
 143                 /* Shows all comments for this item */
 144                 $list = getComments($_type, $ownerId);
 145 
 146                 echo '<div class="comment_header" onclick="toggle_element_by_name(\'comments_holder\')">'.count($list).' comment'.(count($list) == 1 ? '':'s').'</div>';
 147 
 148                 echo '<div id="comments_holder">';
 149                 echo '<div id="comments_only">';
 150                 foreach ($list as $row) {
 151                         echo '<div class="comment_details">';
 152                         echo nameLink($row['userId'], $row['userName']).'<br/>';
 153                         echo $row['timeCreated'];
 154                         echo '</div>';
 155                         echo '<div class="comment_text">'.$row['commentText'];
 156                         if ($session->id) {
 157                                 echo ' | <a href="'.URLadd('delete', $row['commentId']).'"><img src="'.$config['core_web_root'].'gfx/icon_delete.png"/></a>';
 158                         }
 159                         echo '</div>';
 160                 }
 161                 echo '</div>'; //id="comments_only"
 162 
 163                 if ( ($session->id && $_type != COMMENT_MODERATION) ||
 164                                 ($_type == COMMENT_IMAGE || $_type == COMMENT_PASTEBIN)
 165                 ) {
 166                         echo '<form method="post" action="">';
 167                         echo '<textarea name="cmt" cols="'.$col_w.'" rows="'.$col_h.'"></textarea><br/>';
 168                         echo '<input type="submit" class="button" value="Add comment"/>';
 169                         echo '</form>';
 170                 }
 171 
 172                 echo '</div>';  //id="comments_holder"
 173 
 174                 return count($list);
 175         }
