 100         //Also adds 'path_only', that returns 'path', but with script/file name removed. /dir/file.php becomes /dir/
 101         //Also adds 'file_ext', that returns the file extension if its a filename in the URL
 102         function nice_parse_url(&$url)
 103         {
 104                 $url = trim($url);
 105                 $url = str_replace('\"', '', $url);
 106 
 107                 $pos = strpos($url, '://');
 108                 if ($pos === false) {
 109                         $url = 'http://'.$url;
 110                 }
 111 
 112                 $arr = parse_url($url);
 113                 if (!isset($arr['scheme'])) {
 114                         echo 'Scheme undefined!! '.$url.'<br>';
 115                         return false;
 116                 }
 117 
 118                 $arr['file_ext'] = '';
 119 
 120                 if (!isset($arr['path'])) {
 121                         $url .= '/';
 122                         $arr = parse_url($url);
 123                         $arr['path_only'] = '/';
 124                 } else {
 125                         $pos = strrpos($arr['path'], '/');      //last position of a /
 126                         $dir = substr($arr['path'], 0, $pos+1);
 127                         $arr['path_only'] = $dir;
 128 
 129                         //Find the file extension, if one exists
 130                         $filename = substr($arr['path'], $pos+1);
 131                         $pos2 = strpos($filename, '.');                 //fixme: är detta korrekt, tänk med en fil med flera punkter i namnet, strrpos() istället?
 132                         if ($pos2 !== false) $arr['file_ext'] = substr($filename, $pos2);
 133                 }
 134 
 135                 if ($arr['scheme'] != 'http' && $arr['scheme'] != 'https') {
 136                         if ($arr['scheme'] != 'javascript') {
 137                                 echo 'Unsupported scheme: <b>'.$arr['scheme'].'</b> in URL <b>'.$url.'</b><br>';
 138                         }
 139                         return false;
 140                 }
 141 
 142                 return $arr;
 143         }

 146         function extract_filenames($data)
 147         {
 148                 //Normalisera HTML koden
 149                 $data = str_replace("\n", " ", $data);
 150                 $data = str_replace("\r", " ", $data);
 151                 $data = str_replace("\t", " ", $data);
 152 
 153                 $files_found = array();
 154 
 155                 do {
 156                         $org_data = $data;
 157                         $data = str_replace("  ", " ", $data);
 158                 } while ($org_data != $data);
 159 
 160                 //Parsa taggar, identifiera relevanta taggar
 161 
 162                 do {
 163                         $pos1 = strpos($data, '<');                                     //Find first opening bracket
 164                         $pos2 = strpos($data, '>', $pos1);      //Find first closing bracket occuring after $pos1
 165 
 166                         if ($pos1 === false || $pos2 === false) break;
 167 
 168                         $tag = trim(substr($data, $pos1+1, $pos2-$pos1-1));
 169 
 170                         $tag_pos = strpos($tag, ' ');
 171                         if ($tag_pos) {
 172                                 $tag_name = strtolower(substr($tag, 0, $tag_pos));
 173 
 174                                 $tag_params_org = substr($tag, $tag_pos+1);
 175                                 if (strpos($tag_params_org, 'javascript:') === false) {
 176                                         $tag_params = parse_html_parameters($tag_params_org);
 177                                 } else {
 178                                         $tag_params = array();
 179                                 }
 180 
 181                         } else {
 182                                 $tag_name = $tag;
 183                                 $tag_params = array();
 184                         }
 185 
 186                         switch ($tag_name)
 187                         {
 188                                 case 'a':                               //<a href="filename.html">
 189                                 case 'area':            //<area href="blog-rss.php" />
 190                                         if (empty($tag_params['href'])) break;
 191 
 192                                         $pos = strpos($tag_params['href'], '#');
 193                                         if ($pos !== false) $tag_params['href'] = substr($tag_params['href'], 0, $pos);
 194                                         if (!$tag_params['href']) break;
 195 
 196                                         if (substr($tag_params['href'], 0, 7) == 'mailto:') break;
 197 
 198                                         if (!in_array($tag_params['href'], $files_found)) {
 199                                                 //echo '* Located new link: '.$tag_params['href'].'<br>';
 200                                                 $files_found[] = $tag_params['href'];
 201                                         }
 202                                         break;
 203 
 204                                 case 'link':                    //<link href="stylesheet.css" rel="stylesheet" type="text/css">
 205                                         break;
 206 
 207                                 case 'script':          //<script src="/inc/global.js" language="javascript" type="text/javascript">
 208                                         break;
 209                         }
 210 
 211                 } while ( $data = trim(substr($data, $pos2)) );
 212 
 213                 return $files_found;
 214         }

 297         function get_http_contents($url, &$errno)
 298         {
 299                 //todo: $site ska nog inte va global, utan en parameter..
 300                 global $db, $site, $config, $http_request_counter;
 301 
 302                 $errno = 0;
 303                 //echo 'get_http_contents('.$url.')<br>';
 304 
 305                 //check if cached version is not out of date
 306                 $q = 'SELECT body FROM tblSpiderCache WHERE url="'.$db->escape($url).'" AND timeCreated >= NOW()-'.$config['spider']['cache_age'];
 307                 $body = $db->getOneItem($q);
 308                 if ($body) {
 309                         echo '<b>Skipping download of '.$url.')</b> - Using cache<br>';
 310                         return $body;
 311                 }
 312 
 313                 $host = nice_parse_url($url);
 314 
 315                 if (!empty($host['file_ext']) && !in_array($host['file_ext'], $config['spider']['allowed_extensions'])) {
 316                         echo '<b>Skipping download of '.$host['file_ext'].'</b> (from '.$url.')<br>';
 317                         return false;
 318                 }
 319 
 320                 echo 'Downloading and parsing '.$url.' ...<br>';
 321 
 322                 if ($http_request_counter >= $config['spider']['max_http_requests']) {
 323                         echo 'Done '.$http_request_counter.' HTTP requests now. Please reload this page to do some more, dont want to overload the target server<br>';
 324                         die;
 325                 }
 326 
 327                 //todo: handle unusual ports, https etc (?)
 328                 $fp = fsockopen($host['host'], 80, $errno, $errstr, 30);
 329                 if (!$fp) {
 330                         echo "$errstr ($errno)<br />\n";
 331                         return false;
 332                 }
 333 
 334                 //fixme: url parametrar ignoreras här ?!?!?!
 335                 $file = $host['path'];
 336 
 337                 $header  = "GET ".$file." HTTP/1.0\r\n";
 338                 $header .= "Host: ".$host['host']."\r\n";
 339                 $header .= "User-Agent: ".$config['spider']['user_agent']."\r\n";
 340                 $header .= "Connection: close\r\n\r\n";
 341                 fwrite($fp, $header);
 342 
 343                 $result = '';
 344                 while (!feof($fp)) {
 345                         $result .= fgets($fp, 128);
 346                 }
 347                 fclose($fp);
 348 
 349                 //Cut off header
 350                 $pos = strpos($result, "\r\n\r\n");
 351                 $header = substr($result, 0, $pos);
 352                 $body = substr($result, $pos);
 353 
 354                 $arr = explode("\r\n", $header);
 355 
 356                 //todo: kanske parsa upp header-elementen mer tillgängliga som keys i en array,om vi ska göra mycket med header-datan
 357                 foreach ($arr as $val) {
 358                         if (substr($val, 0, 9) == 'HTTP/1.1 ') $errno = intval(substr($val, 9));
 359                 }
 360 
 361                 if ($header && $errno == 0) {
 362                         d($header);
 363                         d($arr);
 364                         die('<b>ERROR! Failed to extract HTTP error number from request for '.$url.'</b><br>');
 365                 }
 366 
 367                 //HTTP/1.1 200 OK
 368                 if ($errno == 200) $errno = 0;
 369 
 370                 //HTTP/1.1 302 Object Moved
 371                 if ($errno == 302)
 372                 {
 373                         $new_loc = '';
 374                         foreach ($arr as $val) {
 375                                 if (substr($val, 0, 10) == 'Location: ') $new_loc = substr($val, 10);
 376                         }
 377 
 378                         if ($new_loc) {
 379                                 echo '<b>302 object moved from '.$url.' to: '.$new_loc.'</b><br>';
 380                                 $site['302'][$url] = $new_loc;
 381                         } else {
 382                                 echo '<b>ERROR: HTTP header 302 object moved didnt specify a new Location:</b><br>';
 383                         }
 384                         $errno = 0;
 385                 }
 386 
 387                 //403 - forbidden
 388                 if ($errno == 403)
 389                 {
 390                         echo '<b>403 - Forbidden: '.$url.'</b><br>';
 391                         $site['403'][$url] = true;
 392                         $errno = 0;
 393                 }
 394 
 395                 //404 - file not found
 396                 if ($errno == 404)
 397                 {
 398                         echo '<b>File not found: '.$url.'</b><br>';
 399                         $site['404'][$url] = true;
 400                         $errno = 0;
 401                 }
 402 
 403                 if (!$errno) {
 404                         //delete old revisions
 405                         $q = 'DELETE FROM tblSpiderCache WHERE url="'.$db->escape($url).'"';
 406                         $db->delete($q);
 407 
 408                         //store header & body in database
 409                         $q = 'INSERT INTO tblSpiderCache SET url="'.$db->escape($url).'",header="'.$db->escape($header).'",body="'.$db->escape($body).'",sha1="'.sha1($body).'",timeCreated=NOW()';
 410                         $db->insert($q);
 411                 } else {
 412                         echo 'errno: '.$errno.'<br>';
 413                 }
 414 
 415                 $http_request_counter++;
 416                 //echo 'done<br/>';
 417 
 418                 return $body;
 419         }

 421         /* Tar en array med absoluta url:er. Genererar en robots.txt utifrån url:erna */
 422         function generate_robots_txt($host, $arr)
 423         {
 424                 $listed = array();
 425 
 426                 $result =
 427                         "# http://".$host."/robots.txt\n".
 428                         "# Generated by crawler at ".date('Y-m-d H:i')."\n".
 429                         "\n".
 430                         "User-agent: *\n";
 431 
 432                 for ($i=0; $i<count($arr); $i++)
 433                 {
 434                         $parsed = nice_parse_url($arr[$i]);
 435 
 436                         if ($parsed['host'] != $host) {
 437                                 echo 'generate_robots_txt() ERROR: IGNORING HOST '.$parsed['host']."\n";
 438                                 continue;
 439                         }
 440 
 441                         $parsed['dir'] = dirname($parsed['path']);
 442                         if ($parsed['dir'] == '\\') $parsed['dir'] = '/';
 443 
 444                         if (!in_array($parsed['dir'], $listed)) {
 445                                 $result .= "Allow: ".$parsed['dir']."\n";
 446 
 447                                 $listed[] = $parsed['dir'];
 448                         }
 449                 }
 450 
 451                 return $result;
 452         }
