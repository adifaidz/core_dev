  10         function addPoll($_type, $ownerId, $text, $duration_mode = '', $start_mode = '')
  11         {
  12                 global $db, $session;
  13 
  14                 if (!is_numeric($_type) || !is_numeric($ownerId)) return false;
  15 
  16                 $text = $db->escape(trim($text));
  17 
  18                 switch ($duration_mode) {
  19                         case 'day';
  20                                 $length = 1;
  21                                 break;
  22 
  23                         case 'week':
  24                                 $length = 7;
  25                                 break;
  26                                 
  27                         case '': break;
  28 
  29                         default: die('eep addpoll');
  30                 }
  31 
  32                 switch ($start_mode) {
  33                         case 'thismonday':
  34                                 $dayofweek = date('N');
  35                                 $mon = date('n');
  36                                 $day = date('j');
  37                                 $thismonday = mktime(6, 0, 0, $mon, $day - $dayofweek + 1);     //06:00 Monday current week
  38                                 $timeStart = ',timeStart="'.sql_datetime($thismonday).'"';
  39                                 $timeEnd = ',timeEnd=DATE_ADD(timeStart, INTERVAL '.$length.' DAY)';
  40                                 break;
  41 
  42                         case 'nextfree':
  43                                 $q = 'SELECT timeEnd FROM tblPolls WHERE pollType='.$_type.' AND ownerId='.$ownerId.' AND deletedBy=0 ORDER BY timeStart DESC LIMIT 1';
  44                                 $data = $db->getOneRow($q);
  45                                 if ($data) {
  46                                         $timeStart = ',timeStart="'.$data['timeEnd'].'"';
  47                                 } else {
  48                                         $timeStart = ',timeStart=NOW()';
  49                                 }
  50                                 $timeEnd = ',timeEnd=DATE_ADD(timeStart, INTERVAL '.$length.' DAY)';
  51                                 break;
  52 
  53                         case '':
  54                                 $timeStart = '';
  55                                 $timeEnd = '';
  56                                 break;
  57 
  58                         default: die('eexp');
  59                 }
  60 
  61                 $q = 'INSERT INTO tblPolls SET ownerId='.$ownerId.',pollType='.$_type.',pollText="'.$text.'",createdBy='.$session->id.',timeCreated=NOW()'.$timeStart.$timeEnd;
  62                 return $db->insert($q);
  63         }

 104         /* Polling gadget */
 105         function poll($_type, $_id)
 106         {
 107                 global $session;
 108                 
 109                 if (!is_numeric($_type) || !is_numeric($_id)) return false;
 110                 
 111                 $data = getPoll($_type, $_id);
 112                 if (!$data) return false;
 113 
 114                 $active = false;
 115                 if (time() >= datetime_to_timestamp($data['timeStart']) && time() <= datetime_to_timestamp($data['timeEnd'])) {
 116                         $active = true;
 117                 }
 118                 if (!$data['timeStart']) $active = true;
 119 
 120                 $result = '<div class="item">';
 121                 if ($active) $result .= 'ACTIVE POLL: ';
 122                 $result .= $data['pollText'].'<br/><br/>';
 123                 $list = getCategories(CATEGORY_POLL, $_id);
 124 
 125                 $result .= '<div id="poll'.$_id.'">';
 126                 if ($session->isAdmin && $data['timeStart']) $result .= 'Starts: '.$data['timeStart'].', ends '.$data['timeEnd'].'<br/>';
 127 
 128                 if ($session->id && $active && !hasAnsweredPoll($_id)) {
 129                         foreach ($list as $row) {
 130                                 $result .= '<span onclick="submit_poll('.$_id.','.$row['categoryId'].')">';
 131                                 $result .= $row['categoryName'];
 132                                 $result .= '</span><br/>';
 133                         }
 134                 } else {
 135                         if ($session->id) {
 136                                 if ($active) {
 137                                         $result .= 'You already voted, showing current standings:<br/>';
 138                                 } else {
 139                                         $result .= 'The poll closed, final result:<br/>';
 140                                 }
 141                         }
 142 
 143                         $votes = getPollStats($_id);
 144                         $tot_votes = 0;
 145                         foreach ($votes as $row) $tot_votes += $row['cnt'];
 146 
 147                         foreach ($votes as $row) {
 148                                 $pct = 0;
 149                                 if ($tot_votes) $pct = (($row['cnt'] / $tot_votes)*100);
 150                                 $result .= $row['categoryName'].' got '.$row['cnt'].' ('.$pct.'%) votes<br/>';
 151                         }
 152                 }
 153 
 154                 if ($session->isAdmin) {
 155                         $result .= '<br/><input type="button" class="button" value="Save as .csv" onclick="get_poll_csv('.$_id.')"/>';
 156                 }
 157 
 158                 $result .= '</div>';
 159 
 160                 if ($session->id) {
 161                         $result .= '<div id="poll_voted'.$_id.'" style="display:none">';
 162                                 $result .= 'Your vote has been registered!';
 163                         $result .= '</div>';
 164                 }
 165 
 166                 $result .= '</div>';    //class="item"
 167 
 168                 return $result;
 169         }

 350         {

 353

 370         function getPollStats($_id)
 371         {
 372                 global $db;
 373                 if (!is_numeric($_id)) return false;
 374 
 375                 $q  = 'SELECT t1.categoryName, ';
 376                 $q .= '(SELECT COUNT(*) FROM tblPollVotes WHERE voteId=t1.categoryId) AS cnt ';
 377                 $q .= 'FROM tblCategories AS t1 ';
 378                 $q .= 'WHERE t1.ownerId='.$_id.' AND t1.categoryType='.CATEGORY_POLL;
