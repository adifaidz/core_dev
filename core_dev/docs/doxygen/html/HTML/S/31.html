 120         /* Returns a input field from the passed data, used together with getUserdataFieldsHTMLEdit() */
 121         function getUserdataInput($row)
 122         {
 123                 global $config;
 124 
 125                 $fieldId = $row['fieldId'];
 126                 if (isset($row['value'])) {
 127                         $value = stripslashes($row['value']);   //doesnt nessecary exist
 128                 } else if (!empty($row['settingValue'])) {
 129                         $value = stripslashes($row['settingValue']);
 130                 } else { //for default values in admin display
 131                         $value = stripslashes($row['fieldDefault']);
 132                 }
 133 
 134                 switch ($row['fieldType']) {
 135                         case USERDATA_TYPE_TEXT:
 136                                 $result = stripslashes($row['fieldName']).': ';
 137                                 $result .= '<input name="userdata_'.$fieldId.'" type="text" value="'.$value.'" size="30" maxlength="50"/>';
 138                                 break;
 139 
 140                         case USERDATA_TYPE_TEXTAREA:
 141                                 $result = stripslashes($row['fieldName']).':<br/>';
 142                                 $result .= '<textarea name="userdata_'.$fieldId.'" rows="6" cols="40">'.$value.'</textarea>';
 143                                 break;
 144 
 145                         case USERDATA_TYPE_CHECKBOX:
 146                                 $result  = '<input name="userdata_'.$fieldId.'" type="hidden" value="0"/>';
 147                                 $result .= '<input name="userdata_'.$fieldId.'" id="userdata_'.$fieldId.'" type="checkbox" class="checkbox" value="1"'.($value == '1'?' checked="checked"':'').'/>';
 148                                 $result .= ' <label for="userdata_'.$fieldId.'">'.$row['fieldName'].'</label>';
 149                                 break;
 150 
 151                         case USERDATA_TYPE_RADIO:
 152                                 $result = stripslashes($row['fieldName']).':<br/>';
 153                                 $options = getCategoriesByOwner(CATEGORY_USERDATA, $fieldId);
 154 
 155                                 foreach($options as $row) {
 156                                         $result .= '<input name="userdata_'.$fieldId.'" type="radio" id="lab_'.$row['categoryId'].'" value="'.$row['categoryId'].'"'.($row['categoryId'] == $value?' checked="checked"':'').'/>';
 157                                         $result .= ' <label for="lab_'.$row['categoryId'].'">'.$row['categoryName'].'</label><br/>';
 158                                 }
 159                                 break;
 160 
 161                         case USERDATA_TYPE_SELECT:
 162                                 $result = stripslashes($row['fieldName']).': ';
 163                                 $result .= getCategoriesSelect(CATEGORY_USERDATA, $fieldId, 'userdata_'.$fieldId, $value);
 164                                 break;
 165 
 166                         case USERDATA_TYPE_IMAGE:
 167                                 $result = stripslashes($row['fieldName']).':<br/>';
 168                                 if ($value) {
 169                                         $result .= makeThumbLink($value);
 170                                         $result .= '<input name="userdata_'.$fieldId.'_remove" id="userdata_'.$fieldId.'_remove" type="checkbox" class="checkbox"/>';
 171                                         $result .= '<label for="userdata_'.$fieldId.'_remove">Delete image</label>';

 173                                         $result .= '<input name="userdata_'.$fieldId.'" type="file"/>';
 174                                 }
 175                                 break;
 176 
 177                         case USERDATA_TYPE_DATE:
 178                                 $result = stripslashes($row['fieldName']).':<br/>';
 179                                 if ($value && (strlen($value) == 8)) {
 180                                         $y = substr($value,0,4);
 181                                         $m = substr($value,4,2);
 182                                         $d = substr($value,6,2);
 183                                 } else {
 184                                         $d = '';
 185                                         $m = '';
 186                                         $y = '';
 187                                 }
 188 
 189                                 $result .= '<select name="userdata_'.$fieldId.'_day">';
 190                                 $result .= '<option value="">- Day -';
 191                                 for ($j=1; $j<=31; $j++) {
 192                                         $k = $j;
 193                                         if ($j<10) $k = '0'.$k;
 194                                         if ($j == $d) $selected = ' selected'; else $selected = '';
 195                                         $result .= '<option value="'.$k.'"'.$selected.'>'.$j;
 196                                 }
 197                                 $result .= '</select>';
 198 
 199                                 $result .= '<select name="userdata_'.$fieldId.'_month">';
 200                                 $result .= '<option value="">- Month -';
 201                                 for ($j=1; $j<=12; $j++) {
 202                                         $k = $j;
 203                                         if ($j<10) $k = '0'.$k;
 204                                         if ($j == $m) $selected = ' selected'; else $selected = '';
 205                                         $result .= '<option value="'.$k.'"'.$selected.'>'.$j;
 206                                 }
 207                                 $result .= '</select>';
 208 
 209                                 $result .= '<select name="userdata_'.$fieldId.'_year">';
 210                                 $result .= '<option value="">- Year -';
 211                                 for ($j=1980; $j<=date('Y'); $j++) {
 212                                         if ($j == $y) $selected = ' selected'; else $selected = '';
 213                                         $result .= '<option value="'.$j.'"'.$selected.'>'.$j;

 216                 }
