 287         function cleanupText($text)
 288         {
 289                 global $db;
 290 
 291                 $text = trim($text);
 292 
 293                 do { /* Remove chunks of whitespace */
 294                         $temp = $text;
 295                         $text = str_replace('  ', ' ', $text);
 296                 } while ($text != $temp);
 297                 
 298                 $text = str_replace('\n', "\n", $text); //ersätter strängen '\n' mot en linefeed
 299                 $text = str_replace('\r', "\r", $text); //ersätter strängen '\r' mot en carriage return
 300 
 301                 $text = str_replace("\r\n", "\n", $text);
 302                 $text = str_replace(" \n", "\n", $text);
