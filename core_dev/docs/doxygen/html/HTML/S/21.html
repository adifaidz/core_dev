 364         /* Returns item data for the last post in any of the threads with parentId=$itemId */
 365         function getForumThreadContentLastPost($itemId)
 366         {
 367                 global $db;
 368                 if (!is_numeric($itemId)) return false;
 369 
 370                 $q = 'SELECT itemId FROM tblForums WHERE parentId='.$itemId.' AND deletedBy=0';;
 371                 $list = $db->getArray($q);
 372 
 373                 $newest_time = 0;
 374 
 375                 for ($i=0; $i<count($list); $i++) {
 376                         $q =    'SELECT itemId, timeCreated FROM tblForums '.
 377                                                 'WHERE parentId='.$list[$i]['itemId'].' AND deletedBy=0 '.
 378                                                 'ORDER BY timeCreated DESC LIMIT 0,1';
 379                         $data = $db->getOneRow($q);
 380 
 381                         if ($data['timeCreated'] > $newest_time) {
 382                                 $newest_time = $data['timeCreated'];
 383                                 $newest_id = $data['itemId'];
 384                         }
 385                 }
 386 
 387                 if ($newest_time) {
 388                         $data = getForumItem($newest_id);
 389                         if (!$data['itemSubject']) {
 390                                 //fills in parent's subject if subject is missing
 391                                 $parent_data = getForumItem($data['parentId']);
 392                                 $data['parentSubject'] = $parent_data['itemSubject'];
 393                         }
 394                         return $data;
 395                 }
 396 
 397                 return false;
 398         }

 677         function getForumSearchResults($criteria, $method, $limit = '')
 678         {
 679                 global $db;
 680 
 681                 if (!$criteria || !$method) return false;
 682 
 683                 $criteria = $db->escape($criteria);
 684

 686 
 687                 $q  = 'SELECT t1.*,t2.userName AS authorName FROM tblForums AS t1 ';
 688                 $q .= 'LEFT JOIN tblUsers AS t2 ON (t1.authorId=t2.userId) ';
 689                 $q .= 'WHERE t1.deletedBy=0 AND '.getForumSearchQuery($list);
 690 
 691                 switch ($method) {
 692                         case 'mostread': //mest läst
 693                                 $q .= 'ORDER BY t1.itemRead DESC '; break;
 694 
 695                         case 'oldfirst': //älst först
 696                                 $q .= 'ORDER BY t1.timeCreated ASC '; break;
 697 
 698                         case 'newfirst': default: //nyast först, default
 699                                 $q .= 'ORDER BY t1.timeCreated DESC '; break;
 700                 }
 701 
 702                 $q .= $limit;
 703
